  130.    * Initiate scroll spy behaviors.

  132.   initScrollSpy() {
  133.     this.gumshoe = new Gumshoe(this.spySelector, {
  134.       navClass: this.spyClass,
  135.       contentClass: this.contentClass,
  136.       offset: this.spyOffset,
  137.       reflow: true,
  138.     });
  140.     document.addEventListener('gumshoeActivate', this.activateScrollSpy, false);
  141.     document.addEventListener(
  142.       'gumshoeDeactivate',
  143.       this.deactivateScrollSpy,
  144.       false,
  145.     );
  147.     if ('IntersectionObserver' in window) {
  148.       const navigationContainer = queryOne(this.containerSelector);
  150.       if (navigationContainer) {
  151.         let previousY = 0;
  152.         let previousRatio = 0;
  153.         let initialized = false;
  155.         this.stickyObserver = new IntersectionObserver(
  156.           (entries) => {
  157.             if (entries && entries[0]) {
  158.               const entry = entries[0];
  159.               const currentY = entry.boundingClientRect.y;
  160.               const currentRatio = entry.intersectionRatio;
  161.               const { isIntersecting } = entry;
  163.               if (!initialized) {
  164.                 initialized = true;
  165.                 previousY = currentY;
  166.                 previousRatio = currentRatio;
  167.                 return;
  168.               }
  170.               if (currentY < previousY) {
  171.                 if (!(currentRatio > previousRatio && isIntersecting)) {
  172.                   // Scrolling down leave
  173.                   this.element.classList.remove(this.spyActiveContainer);
  174.                 }
  175.               } else if (currentY > previousY && isIntersecting) {
  176.                 if (currentRatio > previousRatio) {
  177.                   // Scrolling up enter
  178.                   this.element.classList.add(this.spyActiveContainer);
  179.                 }
  180.               }
  182.               previousY = currentY;
  183.               previousRatio = currentRatio;
  184.             }
  185.           },
  186.           { root: null },
  187.         );
  189.         // observing a target element
  190.         this.stickyObserver.observe(navigationContainer);
  191.       }
  192.     }
  193.   }

  241.   initObserver() {
  242.     if ('MutationObserver' in window) {
  243.       const self = this;
  244.       this.observer = new MutationObserver((mutationsList) => {
  245.         const body = queryOne('.ecl-col-l-9');
  246.         const currentInpage = queryOne('[data-ecl-inpage-navigation-list]');
  248.         mutationsList.forEach((mutation) => {
  249.           // Exclude the changes we perform.
  250.           if (
  251.             mutation &&
  252.             mutation.target &&
  253.             mutation.target.classList &&
  254.             !mutation.target.classList.contains(
  255.               'ecl-inpage-navigation__trigger-current',
  256.             )
  257.           ) {
  258.             // Added nodes.
  259.             if (mutation.addedNodes.length > 0) {
  260.               [].slice.call(mutation.addedNodes).forEach((addedNode) => {
  261.                 if (addedNode.tagName === 'H2' && addedNode.id) {
  262.                   const H2s = queryAll('h2[id]', body);
  263.                   const addedNodeIndex = H2s.findIndex(
  264.                     (H2) => H2.id === addedNode.id,
  265.                   );
  266.                   const element =
  267.                     currentInpage.childNodes[addedNodeIndex - 1].cloneNode(
  268.                       true,
  269.                     );
  270.                   element.childNodes[0].textContent = addedNode.textContent;
  271.                   element.childNodes[0].href = `#${addedNode.id}`;
  272.                   currentInpage.childNodes[addedNodeIndex - 1].after(element);
  273.                 }
  274.               });
  275.             }
  276.             // Removed nodes.
  277.             if (mutation.removedNodes.length > 0) {
  278.               [].slice.call(mutation.removedNodes).forEach((removedNode) => {
  279.                 if (removedNode.tagName === 'H2' && removedNode.id) {
  280.                   currentInpage.childNodes.forEach((item) => {
  281.                     if (
  282.                       item.childNodes[0].href.indexOf(removedNode.id) !== -1
  283.                     ) {
  284.                       // Remove the element from the inpage.
  285.                       item.remove();
  286.                     }
  287.                   });
  288.                 }
  289.               });
  290.             }
  292.             self.update();
  293.           }
  294.         });
  295.       });
  297.       this.observer.observe(document, {
  298.         subtree: true,
  299.         childList: true,
  300.       });
  301.     }
  302.   }

  410.   /**
  411.    * Invoke event listeners on toggle click.
  412.    *
  413.    * @param {Event} e
  414.    */
  415.   handleClickOnToggler(e) {
  416.     const togglerElement = queryOne(this.toggleSelector, this.element);
  418.     e.preventDefault();
  420.     // Get current status
  421.     this.isExpanded = togglerElement.getAttribute('aria-expanded') === 'true';
  423.     // Toggle the expandable/collapsible
  424.     togglerElement.setAttribute(
  425.       'aria-expanded',
  426.       this.isExpanded ? 'false' : 'true',
  427.     );
  428.     if (this.isExpanded) {
  429.       // Untrap focus
  430.       this.focusTrap.deactivate();
  431.     } else {
  432.       // Trap focus
  433.       this.focusTrap.activate();
  434.     }
  436.     this.trigger('onToggle', { isExpanded: this.isExpanded });
  437.   }

  440.    * Sets the necessary attributes to collapse inpage navigation list.

  472.       e.preventDefault();

  474.       if (
  475.         prevItem &&

  479.         if (prevLink) {

  482.       }
  483.     }
  485.     if (e.key === 'ArrowDown') {
  486.       e.preventDefault();
  487.       const nextItem = element.parentElement.nextSibling;
  488.       if (
  489.         nextItem &&
  490.         nextItem.classList.contains('ecl-inpage-navigation__item')
  491.       ) {
  492.         const nextLink = queryOne(this.linksSelector, nextItem);
  493.         if (nextLink) {
  494.           nextLink.focus();
  495.         }
  496.       }
  497.     }
  498.   }
