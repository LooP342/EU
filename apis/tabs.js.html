  526.     const startTab = direction === 'next' ? this.firstTab : this.lastTab;
  527.     const endTab = direction === 'next' ? this.lastTab : this.firstTab;
  529.     if (this.isMobile) {
  530.       if (currentTab !== endTab) {

  532.         this.shiftTabs(direction);
  533.       }
  534.       return;
  535.     }
  537.     if (currentTab === endTab) {
  538.       this.moveFocus(startTab);
  539.     } else {
  540.       this.moveFocus(this.tabsKey[index]);
  541.     }
  542.   }
