  135.     // Get position from center
  136.     const percentOfRange =
  137.       this.rangeInput.value / (this.rangeInput.max - this.rangeInput.min);
  138.     const valuePxPosition = percentOfRange * rect.width;
  139.     const distFromCenter = valuePxPosition - center;
  140.     const percentDistFromCenter = distFromCenter / center;
  142.     // Calculate bubble position
  143.     const offset = percentDistFromCenter * halfThumbWidth;
  144.     const left = rect.left + valuePxPosition - halfLabelWidth - offset;
  146.     this.bubble.style.left = `${left}px`;
