  190.   /**
  191.    * Static method to create a checkbox element.
  192.    *
  193.    * @static
  194.    * @param {Object} options
  195.    * @param {String} options.id
  196.    * @param {String} options.text
  197.    * @param {String} [options.extraClass] - additional CSS class
  198.    * @param {String} [options.disabled] - relevant when re-creating an option
  199.    * @param {String} [options.selected] - relevant when re-creating an option
  200.    * @param {String} ctx
  201.    * @private
  202.    * @returns {HTMLElement}
  203.    */
  204.   static #createCheckbox(options, ctx) {
  205.     // Early returns.
  206.     if (!options || !ctx) return '';
  207.     const { id, text, disabled, selected, extraClass } = options;
  208.     if (!id || !text) return '';
  210.     // Elements to work with.
  211.     const checkbox = document.createElement('div');
  212.     const input = document.createElement('input');
  213.     const label = document.createElement('label');
  214.     const box = document.createElement('span');
  215.     const labelText = document.createElement('span');
  217.     // Respect optional input parameters.
  218.     if (extraClass) {
  219.       checkbox.classList.add(extraClass);
  220.     }
  221.     if (selected) {
  222.       input.setAttribute('checked', true);
  223.     }
  224.     if (disabled) {
  225.       checkbox.classList.add('ecl-checkbox--disabled');
  226.       box.classList.add('ecl-checkbox__box--disabled');
  227.       input.setAttribute('disabled', disabled);
  228.     }
  230.     // Imperative work follows.
  231.     checkbox.classList.add('ecl-checkbox');
  232.     checkbox.setAttribute('data-select-multiple-value', text);
  233.     input.classList.add('ecl-checkbox__input');
  234.     input.setAttribute('type', 'checkbox');
  235.     input.setAttribute('id', `${ctx}-${id}`);
  236.     input.setAttribute('name', `${ctx}-${id}`);
  237.     checkbox.appendChild(input);
  238.     label.classList.add('ecl-checkbox__label');
  239.     label.setAttribute('for', `${ctx}-${id}`);
  240.     box.classList.add('ecl-checkbox__box');
  241.     box.setAttribute('aria-hidden', true);
  242.     box.appendChild(
  243.       Select.#createSvgIcon(
  244.         iconSvgAllCheck,
  245.         'ecl-icon ecl-icon--s ecl-checkbox__icon',
  246.       ),
  247.     );
  248.     label.appendChild(box);
  249.     labelText.classList.add('ecl-checkbox__label-text');
  250.     labelText.innerHTML = text;
  251.     label.appendChild(labelText);
  252.     checkbox.appendChild(label);
  253.     return checkbox;
  254.   }

  869.    *
  870.    * @param {upOrDown}
  871.    * @param {loop}
  872.    * @private

  874.   #moveFocus(upOrDown) {
  875.     const activeEl = document.activeElement;

  890.     const activeIndex = options.indexOf(activeEl);

  896.         );

  905.           this.dropDownToolbar.firstChild.focus();

  907.       }
  908.     } else {
  909.       const previousSiblings = options
  910.         .splice(0, activeIndex)
  911.         .filter(
  912.           (el) => !el.disabled && el.parentElement.style.display !== 'none',
  913.         );
  914.       if (previousSiblings.length > 0) {
  915.         previousSiblings[previousSiblings.length - 1].focus();
  916.       } else {
  917.         this.optionsContainer.scrollTop = 0;
  918.         if (this.selectAll && !this.selectAll.querySelector('input').disabled) {
  919.           this.selectAll.querySelector('input').focus();

  921.           this.search.focus();
